def resolve(resource, position:)
  case resource
  when RDF::Query::Variable then resource
  when RDF::Literal
    case position
    when :subject
      as_literal(Math.cosh(resource.as_number.object))
    when :object
      as_literal(Math.acosh(resource.as_number.object))
    end
  else
    nil
  end
end
