def resolve(resource, position:)
  case position
  when :subject
    return nil unless resource.literal?
    resource = resource.as_datetime
    RDF::Literal(resource.object.strftime("%Y").to_i)
  when :object
    return nil unless resource.literal? || resource.variable?
    resource
  end
end
